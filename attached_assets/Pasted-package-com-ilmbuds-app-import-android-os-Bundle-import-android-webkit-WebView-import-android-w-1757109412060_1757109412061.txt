class MainActivity : ComponentActivity() {
    private var mInterstitialAd: InterstitialAd? = null
    private var mRewardedAd: RewardedAd? = null

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()

        // Initialize AdMob
        MobileAds.initialize(this) {}

        // Load ads
        loadInterstitialAd()
        loadRewardedAd()

        setContent {
            ILMBUDSappTheme {
                MainScreen()
            }
        }
    }

    private fun loadInterstitialAd() {
        val adRequest = AdRequest.Builder().build()

        InterstitialAd.load(
            this,
            "ca-app-pub-9746293142643974/7649626393", // Your Interstitial Ad Unit ID
            adRequest,
            object : InterstitialAdLoadCallback() {
                override fun onAdLoaded(interstitialAd: InterstitialAd) {
                    mInterstitialAd = interstitialAd
                }
            }
        )
    }

    private fun loadRewardedAd() {
        val adRequest = AdRequest.Builder().build()

        RewardedAd.load(
            this,
            "ca-app-pub-9746293142643974/2411518252", // Your Rewarded Ad Unit ID
            adRequest,
            object : RewardedAdLoadCallback() {
                override fun onAdLoaded(rewardedAd: RewardedAd) {
                    mRewardedAd = rewardedAd
                }
            }
        )
    }

    fun showInterstitialAd() {
        mInterstitialAd?.show(this)
        loadInterstitialAd() // Load next ad
    }

    fun showRewardedAd() {
        mRewardedAd?.show(this) { rewardItem ->
            // User earned reward - 10 Quiz Points!
            val rewardAmount = rewardItem.amount
            val rewardType = rewardItem.type
            // Here you can add points to user's account
        }
        loadRewardedAd() // Load next ad
    }
}

// Rest of the code stays the same...
@Composable
fun MainScreen() {
    Column(modifier = Modifier.fillMaxSize()) {
        // WebView takes most space
        WebViewScreen(
            modifier = Modifier
                .fillMaxWidth()
                .weight(1f)
        )

        // Banner Ad at bottom
        BannerAdView(
            modifier = Modifier
                .fillMaxWidth()
                .height(60.dp)
        )
    }
}

@Composable
fun WebViewScreen(modifier: Modifier = Modifier) {
    AndroidView(
        factory = { context ->
            WebView(context).apply {
                layoutParams = ViewGroup.LayoutParams(
                    ViewGroup.LayoutParams.MATCH_PARENT,
                    ViewGroup.LayoutParams.MATCH_PARENT
                )

                settings.apply {
                    javaScriptEnabled = true
                    domStorageEnabled = true
                    loadWithOverviewMode = true
                    useWideViewPort = true
                    setSupportZoom(false)
                    builtInZoomControls = false
                    displayZoomControls = false
                    mixedContentMode = WebSettings.MIXED_CONTENT_ALWAYS_ALLOW
                    cacheMode = WebSettings.LOAD_NO_CACHE
                }

                webViewClient = WebViewClient()
                webChromeClient = WebChromeClient()

                loadUrl("https://ilmbuds-agron6922.replit.app/")
            }
        },
        modifier = modifier
    ) { webView ->
        webView.reload()
    }
}

@Composable
fun BannerAdView(modifier: Modifier = Modifier) {
    val context = LocalContext.current

    AndroidView(
        factory = { context ->
            AdView(context).apply {
                adUnitId = "ca-app-pub-9746293142643974/3548505956" // Your Banner Ad Unit ID
                setAdSize(AdSize.BANNER)

                val adRequest = AdRequest.Builder().build()
                loadAd(adRequest)
            }
        },
        modifier = modifier
    )
}
